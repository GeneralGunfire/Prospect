# Plan 00-02: AppContext, Data Saver Mode, Mobile Layout, Data Loaders

## Objective

Build the application shell: two React contexts (AppContext for globals, QuizContext for quiz state), a mobile-first layout with bottom navigation, a data saver toggle that persists to localStorage, and the typed data loader layer that is the only place `fetch()` is called in the entire app. After this plan, the app has a working shell that subsequent phases can plug into.

## must_haves

- `dataSaverMode` toggle works and survives page reload (localStorage)
- Layout renders correctly at 320px width with no horizontal overflow
- `src/data/loaders.ts` is the only file that calls `fetch()` — all data access goes through it
- AppContext and QuizContext are separate — not merged into one context
- Supabase is NOT imported anywhere in this plan (deferred to Phase 6)

```typescript
import { createContext, useContext, useEffect, useState, ReactNode } from 'react'

type Province =
  | 'Gauteng' | 'Western Cape' | 'KwaZulu-Natal' | 'Eastern Cape'
  | 'Limpopo' | 'Mpumalanga' | 'North West' | 'Free State' | 'Northern Cape'
  | null

interface AppContextValue {
  dataSaverMode: boolean
  toggleDataSaverMode: () => void
  selectedProvince: Province
  setSelectedProvince: (p: Province) => void
}

const AppContext = createContext<AppContextValue | null>(null)

const DATA_SAVER_KEY = 'sa_career_data_saver'

export function AppProvider({ children }: { children: ReactNode }) {
  const [dataSaverMode, setDataSaverMode] = useState<boolean>(() => {
    try {
      return localStorage.getItem(DATA_SAVER_KEY) === 'true'
    } catch {
      return false
    }
  })
  const [selectedProvince, setSelectedProvince] = useState<Province>(null)

  useEffect(() => {
    try {
      localStorage.setItem(DATA_SAVER_KEY, String(dataSaverMode))
    } catch {
      // localStorage unavailable — continue without persistence
    }
  }, [dataSaverMode])

  const toggleDataSaverMode = () => setDataSaverMode(prev => !prev)

  return (
    <AppContext.Provider value={{ dataSaverMode, toggleDataSaverMode, selectedProvince, setSelectedProvince }}>
      {children}
    </AppContext.Provider>
  )
}

export function useApp() {
  const ctx = useContext(AppContext)
  if (!ctx) throw new Error('useApp must be used within AppProvider')
  return ctx
}
```

```typescript
import { createContext, useContext, useState, ReactNode } from 'react'

export type RiasecCode = 'R' | 'I' | 'A' | 'S' | 'E' | 'C'

export interface RiasecScores {
  R: number; I: number; A: number; S: number; E: number; C: number
}

export interface ApsSubject {
  name: string
  mark: number
  points: number // 1-7 scale
}

export interface QuizState {
  riasecScores: RiasecScores | null
  topCodes: RiasecCode[]       // top 3 RIASEC codes ordered
  apsScore: number | null
  apsSubjects: ApsSubject[]
  nsfasEligible: boolean | null
}

interface QuizContextValue {
  quiz: QuizState
  setRiasecResult: (scores: RiasecScores, topCodes: RiasecCode[]) => void
  setApsResult: (score: number, subjects: ApsSubject[]) => void
  setNsfasEligibility: (eligible: boolean) => void
  resetQuiz: () => void
}

const initialQuiz: QuizState = {
  riasecScores: null,
  topCodes: [],
  apsScore: null,
  apsSubjects: [],
  nsfasEligible: null,
}

const QuizContext = createContext<QuizContextValue | null>(null)

export function QuizProvider({ children }: { children: ReactNode }) {
  const [quiz, setQuiz] = useState<QuizState>(initialQuiz)

  const setRiasecResult = (scores: RiasecScores, topCodes: RiasecCode[]) =>
    setQuiz(prev => ({ ...prev, riasecScores: scores, topCodes }))

  const setApsResult = (score: number, subjects: ApsSubject[]) =>
    setQuiz(prev => ({ ...prev, apsScore: score, apsSubjects: subjects }))

  const setNsfasEligibility = (eligible: boolean) =>
    setQuiz(prev => ({ ...prev, nsfasEligible: eligible }))

  const resetQuiz = () => setQuiz(initialQuiz)

  return (
    <QuizContext.Provider value={{ quiz, setRiasecResult, setApsResult, setNsfasEligibility, resetQuiz }}>
      {children}
    </QuizContext.Provider>
  )
}

export function useQuiz() {
  const ctx = useContext(QuizContext)
  if (!ctx) throw new Error('useQuiz must be used within QuizProvider')
  return ctx
}
```

Note: Zustand with persist is added in Phase 1 for the full quiz state machine. This context is the inter-phase communication layer (other features read quiz results from here).

Create `src/data/loaders.ts` — the ONLY file that calls fetch():

```typescript
import type {
  CareerIndexEntry, Career, University, TvetCollege, Bursary, StudyTopic, ProvinceJobDemand
} from './types'

const BASE = '/data'

async function fetchJSON<T>(path: string): Promise<T> {
  const res = await fetch(`${BASE}/${path}`)
  if (!res.ok) throw new Error(`Failed to load ${path}: ${res.status}`)
  return res.json() as Promise<T>
}

// Career index — loaded once, cached by TanStack Query
export const loadCareerIndex = () =>
  fetchJSON<CareerIndexEntry[]>('_master-index.json')

// Individual career — lazy-loaded per career page
export const loadCareer = (pathway: string, id: string) =>
  fetchJSON<Career>(`careers/${pathway}/${id}.json`)

// Reference data — loaded once each
export const loadUniversities = () =>
  fetchJSON<University[]>('universities.json')

export const loadTvetColleges = () =>
  fetchJSON<TvetCollege[]>('tvet-colleges.json')

export const loadBursaries = () =>
  fetchJSON<Bursary[]>('bursaries.json')

// Study content — lazy-loaded per subject+grade
export const loadStudyTopics = (subject: string, grade: 10 | 11 | 12) =>
  fetchJSON<StudyTopic[]>(`study/grade-${grade}/${subject}.json`)

// Job market data
export const loadJobGaps = () =>
  fetchJSON<ProvinceJobDemand[]>('job-gaps.json')
```
