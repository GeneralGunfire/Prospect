# Plan 01-01: RIASEC Quiz State Machine + Quiz UI

## Objective

Build the RIASEC interest quiz as a pure TypeScript state machine (zero React dependency) with a Zustand store for persistence. The quiz asks 42 Likert-scale questions (7 per RIASEC type), scores them, and outputs the top-3 RIASEC code. Results are serialised to a URL param (`?r=RIC&aps=0`) so they are shareable before the APS step.

## must_haves

- Exactly 42 questions (7 per type: R, I, A, S, E, C)
- All 6 RIASEC scores stored separately — never collapsed prematurely
- Top-3 code is ordered (e.g. "RIC" not "CIR") by descending score
- Quiz state persists in localStorage via Zustand — survives accidental navigation on mobile
- Result URL contains `?r=RIC` — parseable by Career Discovery (Phase 3)

```typescript
import type { RiasecCode } from '@/contexts/QuizContext'
import { RIASEC_QUESTIONS } from './riasec-questions'

export type LikertScore = 1 | 2 | 3 | 4 | 5

export interface QuizMachineState {
  currentIndex: number          // 0-41
  answers: Record<string, LikertScore>  // questionId → score
  isComplete: boolean
}

export const INITIAL_STATE: QuizMachineState = {
  currentIndex: 0,
  answers: {},
  isComplete: false,
}

// Pure functions — no side effects, fully testable

export function answerQuestion(
  state: QuizMachineState,
  questionId: string,
  score: LikertScore
): QuizMachineState {
  const answers = { ...state.answers, [questionId]: score }
  const nextIndex = state.currentIndex + 1
  const isComplete = nextIndex >= RIASEC_QUESTIONS.length
  return {
    currentIndex: isComplete ? state.currentIndex : nextIndex,
    answers,
    isComplete,
  }
}

export function goBack(state: QuizMachineState): QuizMachineState {
  if (state.currentIndex === 0) return state
  return { ...state, currentIndex: state.currentIndex - 1, isComplete: false }
}

export function calculateScores(answers: Record<string, LikertScore>): Record<RiasecCode, number> {
  const scores: Record<RiasecCode, number> = { R: 0, I: 0, A: 0, S: 0, E: 0, C: 0 }
  for (const q of RIASEC_QUESTIONS) {
    scores[q.type] += answers[q.id] ?? 0
  }
  return scores
}

export function getTopCodes(scores: Record<RiasecCode, number>): RiasecCode[] {
  return (Object.entries(scores) as [RiasecCode, number][])
    .sort(([, a], [, b]) => b - a)
    .slice(0, 3)
    .map(([code]) => code)
}

export function getProgress(state: QuizMachineState): number {
  return Math.round((state.currentIndex / RIASEC_QUESTIONS.length) * 100)
}
```

Create `src/features/quiz/useQuizMachine.ts` (Zustand store wrapping the state machine):

```typescript
import { create } from 'zustand'
import { persist } from 'zustand/middleware'
import {
  QuizMachineState, INITIAL_STATE, LikertScore,
  answerQuestion, goBack, calculateScores, getTopCodes
} from './quiz.machine'
import type { RiasecCode, RiasecScores } from '@/contexts/QuizContext'
import { RIASEC_QUESTIONS } from './riasec-questions'

interface QuizStore extends QuizMachineState {
  answer: (questionId: string, score: LikertScore) => void
  back: () => void
  reset: () => void
  getCurrentQuestion: () => typeof RIASEC_QUESTIONS[number] | null
  getResults: () => { scores: RiasecScores; topCodes: RiasecCode[] } | null
}

export const useQuizMachine = create<QuizStore>()(
  persist(
    (set, get) => ({
      ...INITIAL_STATE,

      answer: (questionId, score) =>
        set(state => answerQuestion(state, questionId, score)),

      back: () => set(state => goBack(state)),

      reset: () => set(INITIAL_STATE),

      getCurrentQuestion: () => {
        const { currentIndex, isComplete } = get()
        if (isComplete) return null
        return RIASEC_QUESTIONS[currentIndex] ?? null
      },

      getResults: () => {
        const { answers, isComplete } = get()
        if (!isComplete) return null
        const scores = calculateScores(answers) as RiasecScores
        const topCodes = getTopCodes(scores)
        return { scores, topCodes }
      },
    }),
    { name: 'sa_career_quiz' }
  )
)
```

```typescript
import { useQuizMachine } from './useQuizMachine'
import { RIASEC_QUESTIONS } from './riasec-questions'
import type { LikertScore } from './quiz.machine'
import { useNavigate } from 'react-router'
import { useQuiz } from '@/contexts/QuizContext'

const LIKERT_LABELS: Record<LikertScore, string> = {
  1: 'Strongly Disagree',
  2: 'Disagree',
  3: 'Neutral',
  4: 'Agree',
  5: 'Strongly Agree',
}

export default function RiasecQuiz() {
  const { currentIndex, answers, isComplete, answer, back, reset, getCurrentQuestion, getResults } = useQuizMachine()
  const { setRiasecResult } = useQuiz()
  const navigate = useNavigate()

  const question = getCurrentQuestion()
  const progress = Math.round((currentIndex / RIASEC_QUESTIONS.length) * 100)

  // Auto-navigate when complete
  if (isComplete) {
    const results = getResults()
    if (results) {
      setRiasecResult(results.scores, results.topCodes)
      const code = results.topCodes.join('')
      navigate(`/quiz/results?r=${code}`, { replace: true })
    }
    return null
  }

  if (!question) return null

  const currentAnswer = answers[question.id]

  return (
    <div className="py-6">
      {/* Progress bar */}
      <div className="mb-6">
        <div className="flex justify-between text-xs text-gray-500 mb-1">
          <span>Question {currentIndex + 1} of {RIASEC_QUESTIONS.length}</span>
          <span>{progress}%</span>
        </div>
        <div className="w-full bg-gray-200 rounded-full h-2">
          <div
            className="bg-blue-600 h-2 rounded-full transition-all duration-300"
            style={{ width: `${progress}%` }}
          />
        </div>
      </div>

      {/* Question */}
      <h2 className="text-lg font-semibold text-gray-900 mb-6 leading-snug">
        {question.text}
      </h2>

      {/* Likert options */}
      <div className="flex flex-col gap-3">
        {([1, 2, 3, 4, 5] as LikertScore[]).map(score => (
          <button
            key={score}
            onClick={() => answer(question.id, score)}
            className={`w-full text-left px-4 py-3 rounded-lg border text-sm font-medium transition-colors ${
              currentAnswer === score
                ? 'bg-blue-600 text-white border-blue-600'
                : 'bg-white text-gray-700 border-gray-300 active:bg-gray-50'
            }`}
          >
            {score}. {LIKERT_LABELS[score]}
          </button>
        ))}
      </div>

      {/* Back button */}
      {currentIndex > 0 && (
        <button
          onClick={back}
          className="mt-6 text-sm text-gray-500 underline"
        >
          ← Back
        </button>
      )}

      {/* Reset */}
      <button
        onClick={() => { reset(); }}
        className="mt-4 ml-4 text-sm text-gray-400 underline"
      >
        Start over
      </button>
    </div>
  )
}
```

import { useSearchParams, Link } from 'react-router'
import { useQuiz } from '@/contexts/QuizContext'
import { RIASEC_TYPE_NAMES, RIASEC_TYPE_DESCRIPTIONS } from './riasec-questions'
import type { RiasecCode } from '@/contexts/QuizContext'

export default function QuizResults() {
  const [searchParams] = useSearchParams()
  const { quiz } = useQuiz()
  const rParam = searchParams.get('r') ?? ''
  const apsParam = searchParams.get('aps')

  const topCodes = (rParam.split('') as RiasecCode[]).slice(0, 3)

  if (topCodes.length === 0) {
    return (
      <div className="py-8 text-center">
        <p className="text-gray-500">No results yet.</p>
        <Link to="/quiz" className="text-blue-600 underline mt-2 block">Take the quiz</Link>
      </div>
    )
  }

  const shareUrl = `${window.location.origin}/quiz/results?r=${rParam}${apsParam ? `&aps=${apsParam}` : ''}`
  const whatsappText = encodeURIComponent(`My career type: ${topCodes.join('')} — ${topCodes.map(c => RIASEC_TYPE_NAMES[c]).join(', ')}. Find your career path: ${shareUrl}`)

  return (
    <div className="py-6">
      <h1 className="text-2xl font-bold text-gray-900 mb-2">Your Career Type</h1>
      <p className="text-gray-500 text-sm mb-6">Based on your interests</p>

      <div className="flex gap-3 mb-6">
        {topCodes.map((code, i) => (
          <div key={code} className={`flex-1 rounded-xl p-4 text-center ${i === 0 ? 'bg-blue-600 text-white' : 'bg-gray-100 text-gray-700'}`}>
            <div className="text-3xl font-black">{code}</div>
            <div className="text-xs mt-1 font-medium">{RIASEC_TYPE_NAMES[code]}</div>
          </div>
        ))}

      <p className="text-gray-700 text-sm mb-6 leading-relaxed">
        {RIASEC_TYPE_DESCRIPTIONS[topCodes[0]]}

      {/* Actions */}
      <div className="flex flex-col gap-3">
        <Link
          to={`/careers?r=${rParam}${apsParam ? `&aps=${apsParam}` : ''}`}
          className="block w-full bg-blue-600 text-white text-center py-3 rounded-lg font-semibold"
        >
          See matching careers →
        </Link>

        <a
          href={`https://wa.me/?text=${whatsappText}`}
          target="_blank"
          rel="noopener noreferrer"
          className="block w-full border border-gray-300 text-gray-700 text-center py-3 rounded-lg font-medium"

      </div>
