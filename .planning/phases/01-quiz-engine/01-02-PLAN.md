# Plan 01-02: APS Calculator + NSFAS Checker + Shareable URL

## Objective

Build the APS calculator (per-subject mark entry → aggregate APS score + subject prerequisite points), NSFAS eligibility checker (rule-based household income check), and the full shareable result URL (`/quiz/results?r=RIC&aps=32`) that carries both RIASEC and APS data to Career Discovery.

## must_haves

- APS points scale matches the standard SA NSC scale (1–7 points per subject)
- Life Orientation is excluded from APS total (standard SA university convention)
- NSFAS threshold is R350,000 (verified 2025) with note about R350K–R600K loan option
- Share URL contains both `r=` and `aps=` params and stays under 200 chars
- Career matches route (`/careers?r=RIC&aps=32`) receives both params

export interface ApsEntry {
  subject: string
  mark: number        // percentage 0-100
}

export interface ApsResult {
  subjects: ApsEntry[]
  totalAps: number
  subjectPoints: Record<string, number>  // subject → 1-7 points
}

// Standard SA NSC APS conversion (used by most public universities)
// Life Orientation is excluded from APS totals
export function markToPoints(mark: number): number {
  if (mark >= 80) return 7
  if (mark >= 70) return 6
  if (mark >= 60) return 5
  if (mark >= 50) return 4
  if (mark >= 40) return 3
  if (mark >= 30) return 2
  return 1
}

export const EXCLUDED_SUBJECTS = ['Life Orientation', 'LO']

export function calculateAps(entries: ApsEntry[]): ApsResult {
  const subjectPoints: Record<string, number> = {}
  let total = 0

  for (const entry of entries) {
    const points = markToPoints(entry.mark)
    subjectPoints[entry.subject] = points
    if (!EXCLUDED_SUBJECTS.includes(entry.subject)) {
      total += points
    }
  }

  return { subjects: entries, totalAps: total, subjectPoints }
}

// Standard matric subject list for the APS calculator form
export const MATRIC_SUBJECTS = [
  'English Home Language',
  'English First Additional Language',
  'Afrikaans Home Language',
  'Afrikaans First Additional Language',
  'IsiZulu Home Language',
  'Mathematics',
  'Mathematical Literacy',
  'Physical Sciences',
  'Life Sciences',
  'Accounting',
  'Business Studies',
  'Economics',
  'Geography',
  'History',
  'Computer Applications Technology',
  'Engineering Graphics and Design',
  'Agricultural Sciences',
  'Life Orientation',
] as const

export type MatricSubject = typeof MATRIC_SUBJECTS[number]
```

import { useState } from 'react'
import { useNavigate, useSearchParams } from 'react-router'
import { useQuiz } from '@/contexts/QuizContext'
import { calculateAps, markToPoints, MATRIC_SUBJECTS, EXCLUDED_SUBJECTS } from './aps.machine'

const DEFAULT_SUBJECTS = [
  'English Home Language',
  'Mathematics',
  'Physical Sciences',
  'Life Sciences',
  'Accounting',
  'Life Orientation',
]

export default function ApsCalculator() {
  const [searchParams] = useSearchParams()
  const navigate = useNavigate()
  const { setApsResult } = useQuiz()
  const rParam = searchParams.get('r') ?? ''

  const [entries, setEntries] = useState<Record<string, string>>(
    Object.fromEntries(DEFAULT_SUBJECTS.map(s => [s, '']))
  )
  const [extraSubject, setExtraSubject] = useState('')

  const filledEntries = Object.entries(entries)
    .filter(([, v]) => v !== '' && !isNaN(Number(v)))
    .map(([subject, mark]) => ({ subject, mark: Number(mark) }))

  const result = filledEntries.length > 0 ? calculateAps(filledEntries) : null

  const handleSubmit = () => {
    if (!result) return
    const subjects = filledEntries.map(e => ({
      name: e.subject,
      mark: e.mark,
      points: markToPoints(e.mark),
    }))
    setApsResult(result.totalAps, subjects)
    navigate(`/quiz/results?r=${rParam}&aps=${result.totalAps}`)
  }

  return (
    <div className="py-6">
      <h1 className="text-2xl font-bold text-gray-900 mb-1">APS Calculator</h1>
      <p className="text-sm text-gray-500 mb-6">Enter your subject marks (%) to calculate your APS score</p>

      <div className="flex flex-col gap-3 mb-6">
        {Object.keys(entries).map(subject => {
          const isExcluded = EXCLUDED_SUBJECTS.includes(subject)
          const points = entries[subject] ? markToPoints(Number(entries[subject])) : null
          return (
            <div key={subject} className="flex items-center gap-3">
              <div className="flex-1 text-sm text-gray-700 leading-tight">{subject}</div>
              <input
                type="number"
                min="0"
                max="100"
                value={entries[subject]}
                onChange={e => setEntries(prev => ({ ...prev, [subject]: e.target.value }))}
                placeholder="%"
                className="w-16 border border-gray-300 rounded px-2 py-1.5 text-sm text-center"
              />
              {points !== null && (
                <span className={`text-xs font-bold w-8 text-right ${isExcluded ? 'text-gray-400' : 'text-blue-600'}`}>
                  {isExcluded ? '—' : `${points}pt`}
                </span>
              )}
            </div>
          )
        })}
      </div>

      {/* APS total */}
      {result && (
        <div className="bg-blue-50 border border-blue-200 rounded-xl p-4 mb-6 text-center">
          <div className="text-4xl font-black text-blue-600">{result.totalAps}</div>
          <div className="text-sm text-gray-600 mt-1">Your APS Score</div>
          <div className="text-xs text-gray-400 mt-0.5">(Life Orientation excluded per university standard)</div>
        </div>
      )}

      <button
        onClick={handleSubmit}
        disabled={!result || result.totalAps === 0}
        className="w-full bg-blue-600 text-white py-3 rounded-lg font-semibold disabled:opacity-40"
      >
        See careers I qualify for →
      </button>
    </div>
  )
}
```

```typescript
import { useState } from 'react'
import { useQuiz } from '@/contexts/QuizContext'

// Verified 2025 — verify annually at nsfas.org.za
const BURSARY_THRESHOLD = 350_000
const LOAN_THRESHOLD = 600_000

type Eligibility = 'bursary' | 'loan' | 'ineligible' | null

export default function NsfasChecker() {
  const [income, setIncome] = useState('')
  const [saResident, setSaResident] = useState<boolean | null>(null)
  const [firstTime, setFirstTime] = useState<boolean | null>(null)
  const { setNsfasEligibility } = useQuiz()

  const householdIncome = income ? Number(income.replace(/\D/g, '')) : null

  const getEligibility = (): Eligibility => {
    if (householdIncome === null || saResident === null || firstTime === null) return null
    if (!saResident || !firstTime) return 'ineligible'
    if (householdIncome <= BURSARY_THRESHOLD) return 'bursary'
    if (householdIncome <= LOAN_THRESHOLD) return 'loan'
    return 'ineligible'
  }

  const eligibility = getEligibility()

  const handleConfirm = () => {
    if (eligibility === null) return
    setNsfasEligibility(eligibility === 'bursary' || eligibility === 'loan')
  }

  return (
    <div className="py-6">
      <h2 className="text-xl font-bold text-gray-900 mb-4">NSFAS Eligibility Check</h2>

      <div className="flex flex-col gap-5">
        {/* Household income */}
        <div>
          <label className="block text-sm font-medium text-gray-700 mb-1">
            Combined household income per year (R)
          </label>
          <input
            type="text"
            inputMode="numeric"
            value={income}
            onChange={e => setIncome(e.target.value)}
            placeholder="e.g. 180000"
            className="w-full border border-gray-300 rounded-lg px-3 py-2.5 text-sm"
          />
        </div>

        {/* SA citizen / permanent resident */}
        <div>
          <p className="text-sm font-medium text-gray-700 mb-2">Are you a South African citizen or permanent resident?</p>
          <div className="flex gap-3">
            {[true, false].map(val => (
              <button key={String(val)} onClick={() => setSaResident(val)}
                className={`flex-1 py-2.5 rounded-lg border text-sm font-medium ${saResident === val ? 'bg-blue-600 text-white border-blue-600' : 'border-gray-300 text-gray-700'}`}>
                {val ? 'Yes' : 'No'}
              </button>
            ))}
          </div>
        </div>

        {/* First-time student */}
        <div>
          <p className="text-sm font-medium text-gray-700 mb-2">Will this be your first time at a university or TVET college?</p>
          <div className="flex gap-3">
            {[true, false].map(val => (
              <button key={String(val)} onClick={() => setFirstTime(val)}
                className={`flex-1 py-2.5 rounded-lg border text-sm font-medium ${firstTime === val ? 'bg-blue-600 text-white border-blue-600' : 'border-gray-300 text-gray-700'}`}>
                {val ? 'Yes' : 'No'}
              </button>
            ))}
          </div>
        </div>
      </div>

      {/* Result */}
      {eligibility === 'bursary' && (
        <div className="mt-6 bg-green-50 border border-green-200 rounded-xl p-4">
          <div className="font-bold text-green-700 mb-1">✓ Likely NSFAS eligible (bursary)</div>
          <p className="text-sm text-green-700">Your household income is below R350,000. You may qualify for a full NSFAS bursary covering tuition, accommodation, and living costs.</p>
          <a href="https://www.nsfas.org.za" target="_blank" rel="noopener noreferrer" className="text-xs text-green-600 underline mt-2 block">Apply at nsfas.org.za →</a>
        </div>
      )}
      {eligibility === 'loan' && (
        <div className="mt-6 bg-yellow-50 border border-yellow-200 rounded-xl p-4">
          <div className="font-bold text-yellow-700 mb-1">~ NSFAS loan may be available</div>
          <p className="text-sm text-yellow-700">Your household income is R350,001–R600,000. You may qualify for a NSFAS loan (not a bursary). Loans must be repaid after graduation.</p>
          <a href="https://www.nsfas.org.za" target="_blank" rel="noopener noreferrer" className="text-xs text-yellow-600 underline mt-2 block">Check at nsfas.org.za →</a>
        </div>
      )}
      {eligibility === 'ineligible' && (
        <div className="mt-6 bg-gray-50 border border-gray-200 rounded-xl p-4">
          <div className="font-bold text-gray-700 mb-1">NSFAS not applicable</div>
          <p className="text-sm text-gray-600">Based on your answers, NSFAS funding does not apply. Check the bursary database for private bursaries.</p>
        </div>
      )}

      {eligibility !== null && (
        <button onClick={handleConfirm} className="mt-4 w-full bg-blue-600 text-white py-3 rounded-lg font-semibold">
          Continue to bursaries →
        </button>
      )}

      <p className="text-xs text-gray-400 mt-4">Thresholds verified for 2025. Check nsfas.org.za annually for updates.</p>
    </div>
  )
}
```
