# Plan 03-01: Career List UI with RIASEC + APS Filtering

## Objective

Build the career discovery page: load the master index via TanStack Query, filter careers by RIASEC codes and APS eligibility, and display them in a card grid. Filters read from URL params so they are shareable and survive page refresh. Province dropdown filter included.

## must_haves

- Career list reads from `public/data/_master-index.json` via TanStack Query (never bundled)
- RIASEC filter reads `?r=` URL param set by quiz — no manual re-entry needed
- APS filter shows `eligible` / `within reach` / `not yet` badges correctly
- Province dropdown filters by `demand_provinces[]` field in career data
- TanStack Query is wired at app root (QueryClient provider)

```typescript
import type { CareerIndexEntry, RiasecCode, Province } from '@/data/types'

export type ApsEligibility = 'eligible' | 'within-reach' | 'not-yet' | 'unknown'

export function getApsEligibility(career: CareerIndexEntry, apsScore: number | null): ApsEligibility {
  if (apsScore === null) return 'unknown'
  if (apsScore >= career.aps_minimum) return 'eligible'
  if (apsScore >= career.aps_minimum - 5) return 'within-reach'
  return 'not-yet'
}

export interface CareerFilters {
  riasecCodes: RiasecCode[]    // from ?r= param
  apsScore: number | null      // from ?aps= param
  apsFilter: ApsEligibility | 'all'
  province: Province | null
  searchQuery: string
}

export function filterCareers(
  careers: CareerIndexEntry[],
  filters: CareerFilters
): CareerIndexEntry[] {
  return careers.filter(career => {
    // RIASEC filter — career must share at least one code with user's top codes
    if (filters.riasecCodes.length > 0) {
      const hasMatch = career.riasec_codes.some(c => filters.riasecCodes.includes(c))
      if (!hasMatch) return false
    }

    // APS eligibility filter
    if (filters.apsFilter !== 'all') {
      const eligibility = getApsEligibility(career, filters.apsScore)
      if (eligibility !== filters.apsFilter) return false
    }

    // Province filter
    if (filters.province) {
      if (!career.demand_provinces.includes(filters.province)) return false
    }

    return true
  })
}

export function sortCareers(
  careers: CareerIndexEntry[],
  riasecCodes: RiasecCode[],
  apsScore: number | null
): CareerIndexEntry[] {
  return [...careers].sort((a, b) => {
    // Score by number of matching RIASEC codes
    const aMatches = a.riasec_codes.filter(c => riasecCodes.includes(c)).length
    const bMatches = b.riasec_codes.filter(c => riasecCodes.includes(c)).length
    if (bMatches !== aMatches) return bMatches - aMatches

    // Then by APS eligibility (eligible first)
    if (apsScore !== null) {
      const aElig = getApsEligibility(a, apsScore)
      const bElig = getApsEligibility(b, apsScore)
      const order: ApsEligibility[] = ['eligible', 'within-reach', 'not-yet', 'unknown']
      return order.indexOf(aElig) - order.indexOf(bElig)
    }

    return 0
  })
}
```

```typescript
import { useSearchParams } from 'react-router'
import type { Province } from '@/data/types'
import type { ApsEligibility } from './filterCareers'

const PROVINCES: Province[] = [
  'Gauteng', 'Western Cape', 'KwaZulu-Natal', 'Eastern Cape',
  'Limpopo', 'Mpumalanga', 'North West', 'Free State', 'Northern Cape'
]

export default function FilterBar() {
  const [params, setParams] = useSearchParams()
  const aps = params.get('aps')
  const currentApsFilter = (params.get('apsFilter') ?? 'all') as ApsEligibility | 'all'
  const currentProvince = params.get('province') as Province | null

  const setFilter = (key: string, value: string | null) => {
    setParams(prev => {
      const next = new URLSearchParams(prev)
      if (value === null || value === 'all' || value === '') {
        next.delete(key)
      } else {
        next.set(key, value)
      }
      return next
    })
  }

  return (
    <div className="flex flex-col gap-2 mb-4">
      {/* APS eligibility filter — only show if APS is known */}
      {aps && (
        <div className="flex gap-2 overflow-x-auto pb-1">
          {(['all', 'eligible', 'within-reach'] as const).map(val => (
            <button
              key={val}
              onClick={() => setFilter('apsFilter', val)}
              className={`shrink-0 text-xs px-3 py-1.5 rounded-full border font-medium ${
                currentApsFilter === val
                  ? 'bg-blue-600 text-white border-blue-600'
                  : 'bg-white text-gray-600 border-gray-300'
              }`}
            >
              {val === 'all' ? 'All careers' : val === 'eligible' ? '✓ I qualify' : '~ Almost there'}
            </button>
          ))}
        </div>
      )}

      {/* Province dropdown */}
      <select
        value={currentProvince ?? ''}
        onChange={e => setFilter('province', e.target.value || null)}
        className="w-full border border-gray-300 rounded-lg px-3 py-2 text-sm text-gray-700 bg-white"
      >
        <option value="">All provinces</option>
        {PROVINCES.map(p => (
          <option key={p} value={p}>{p}</option>
        ))}
      </select>
    </div>
  )
}
```

Create `src/features/careers/CareerList.tsx`:

```typescript
import { useSearchParams } from 'react-router'
import { useCareerIndex } from './useCareerIndex'
import { filterCareers, sortCareers, type CareerFilters } from './filterCareers'
import CareerCard from './CareerCard'
import FilterBar from './FilterBar'
import type { RiasecCode, Province } from '@/data/types'

export default function CareerList() {
  const [params] = useSearchParams()
  const { data: careers, isLoading, error } = useCareerIndex()

  const rParam = params.get('r') ?? ''
  const apsParam = params.get('aps')
  const apsScore = apsParam ? Number(apsParam) : null
  const riasecCodes = rParam.split('').filter(Boolean) as RiasecCode[]
  const apsFilter = (params.get('apsFilter') ?? 'all') as CareerFilters['apsFilter']
  const province = params.get('province') as Province | null

  if (isLoading) return <div className="py-8 text-center text-gray-400 text-sm">Loading careers...</div>
  if (error) return <div className="py-8 text-center text-red-500 text-sm">Failed to load careers. Check your connection.</div>
  if (!careers) return null

  const filters: CareerFilters = { riasecCodes, apsScore, apsFilter, province, searchQuery: '' }
  const filtered = filterCareers(careers, filters)
  const sorted = sortCareers(filtered, riasecCodes, apsScore)

  return (
    <div className="py-6">
      <div className="flex items-center justify-between mb-2">
        <h1 className="text-xl font-bold text-gray-900">
          {riasecCodes.length > 0 ? `Careers for ${riasecCodes.join('')}` : 'All Careers'}
        </h1>
        <span className="text-xs text-gray-400">{sorted.length} found</span>
      </div>

      {apsScore && (
        <p className="text-sm text-gray-500 mb-4">APS score: <span className="font-bold text-blue-600">{apsScore}</span></p>
      )}

      <FilterBar />

      {sorted.length === 0 ? (
        <div className="py-8 text-center text-gray-400 text-sm">
          No careers match your filters.
          <button onClick={() => window.history.back()} className="block mx-auto mt-2 text-blue-600 underline">Go back</button>
        </div>
      ) : (
        <div className="flex flex-col gap-3">
          {sorted.map(career => (
            <CareerCard key={career.id} career={career} apsScore={apsScore} />
          ))}
        </div>
      )}
    </div>
  )
}
```
