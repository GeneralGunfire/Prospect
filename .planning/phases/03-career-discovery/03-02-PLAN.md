# Plan 03-02: Fuzzy Search + URL Param Persistence

## Objective

Add Fuse.js-powered fuzzy search to the career list so students can type "nurse", "code", or "electric" and get relevant results. All filters (RIASEC, APS, province, search query) must be driven by URL params so the state is shareable and bookmarkable.

## must_haves

- Fuse.js search over career `title`, `category`, and `riasec_codes`
- Debounced input (300ms) to avoid re-querying on every keystroke
- All active filters (r, aps, apsFilter, province, q) live in URL params — no local state for filter values
- Clearing search restores previous results instantly (no loading state)
- `npm run build` exits 0

```typescript
import Fuse from 'fuse.js'
import { useMemo } from 'react'
import type { CareerIndexEntry } from '@/data/types'

const FUSE_OPTIONS: Fuse.IFuseOptions<CareerIndexEntry> = {
  keys: [
    { name: 'title', weight: 0.6 },
    { name: 'category', weight: 0.2 },
    { name: 'riasec_codes', weight: 0.2 },
  ],
  threshold: 0.35,      // 0 = exact, 1 = match anything — 0.35 is a good balance
  minMatchCharLength: 2,
  includeScore: false,
}

export function useCareerSearch(
  careers: CareerIndexEntry[],
  query: string
): CareerIndexEntry[] {
  const fuse = useMemo(() => new Fuse(careers, FUSE_OPTIONS), [careers])

  return useMemo(() => {
    const q = query.trim()
    if (q.length < 2) return careers
    return fuse.search(q).map(r => r.item)
  }, [fuse, query, careers])
}
```

### Task 3: Create SearchBar component

<action>
Create `src/features/careers/SearchBar.tsx`:

```typescript
import { useSearchParams } from 'react-router'
import { useCallback, useEffect, useState } from 'react'

export default function SearchBar() {
  const [searchParams, setSearchParams] = useSearchParams()
  const [localValue, setLocalValue] = useState(searchParams.get('q') ?? '')

  // Debounce: update URL only after 300ms idle
  useEffect(() => {
    const timer = setTimeout(() => {
      setSearchParams(prev => {
        const next = new URLSearchParams(prev)
        if (localValue.trim()) {
          next.set('q', localValue.trim())
        } else {
          next.delete('q')
        }
        return next
      }, { replace: true })
    }, 300)
    return () => clearTimeout(timer)
  }, [localValue, setSearchParams])

  const handleClear = useCallback(() => {
    setLocalValue('')
  }, [])

  return (
    <div className="relative">
      <input
        type="search"
        value={localValue}
        onChange={e => setLocalValue(e.target.value)}
        placeholder="Search careers…"
        aria-label="Search careers"
        className="w-full border border-gray-300 rounded-lg px-3 py-2.5 pr-8 text-sm focus:outline-none focus:ring-2 focus:ring-blue-500"
      />
      {localValue && (
        <button
          onClick={handleClear}
          aria-label="Clear search"
          className="absolute right-2.5 top-1/2 -translate-y-1/2 text-gray-400 hover:text-gray-600 text-lg leading-none"
        >
          ×
        </button>
      )}
    </div>
  )
}
```
